nsc_flagship_bow = {
	entity = "flagship_construction_yard_entity"
	construction_entity = "flagship_construction_yard_entity"
	portrait = GFX_megastructure_construction_background
	place_entity_on_planet_plane = no
	entity_offset = { x = @nsc_flagship_entity_offset_x y = @nsc_flagship_entity_offset_y }
	build_time = 250
	resources = {
		category = megastructures
		cost = {
			alloys = 4000
		}
		cost = {
			alloys = 40
			multiplier = event_target:nsc_modmenu_data_storage.nsc_modmenu_ship_class_Flagship_build_cost_var
		}
		produces = {
			influence = 3
			unity = 25
		}
		upkeep = {
			energy = 50
			alloys = 50
		}
		upkeep = {
			energy = 0.5
			alloys = 0.5
			multiplier = event_target:nsc_modmenu_data_storage.nsc_modmenu_ship_class_Flagship_build_upkeep_var
		}
	}

	upgrade_from = {
		nsc_flagship_shipyard
	}

	possible = {
		nsc_modmenu_game_content_on_trigger = { CATEGORY = ship SUBCAT = class ID = Flagship }
	}

	prerequisites = { "tech_Flagship_1" }

	on_build_complete = {
		remove_megastructure = fromfrom
		spawn_megastructure = {
			type = nsc_flagship_shipyard_complete
			owner = from
		}
	}
}
