    potential = {
        has_global_flag = nsc_modmenu_ship_class_page_opened
        nsc_modmenu_game_content_on_trigger = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser }
    }

    effect = {
        custom_tooltip = nsc_modmenu_game_content_disable_effect_tooltip

    }
}

        custom_tooltip = nsc_modmenu_game_content_enable_effect_tooltip

    }
}

    }
    effect = {
        custom_tooltip = nsc_modmenu_ship_class_build_cost_down_effect_tooltip
        hidden_effect = {
            nsc_modmenu_game_content_change_variable_effect = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser MODIFIER = build_cost VALUE = -5 }
            every_playable_country = { nsc_modmenu_game_content_add_modifier_effect = { CATEGORY = ship SUBCAT = class ID = StrikeCruiser MODIFIER = build_cost } }
        }
    }
}
